enum RoleEnum {
  AGENT
  SUPERVISOR
  ADM
}


model User {
  id            String   @id @default(uuid())
  name          String
  email         String   @unique
  passwordHash  String
  phoneNumber   String
  role          RoleEnum
  banned        Boolean  @default(false)
  createdAt     DateTime @default(now())

  // ===== AGENT =====
  block         String?
  registration  String?

  // ===== SUPERVISOR =====
  matricula     String?

  @@index([role])
}


generator client {
  provider = "prisma-client"
  output   = "../src/generated/prisma"
}

datasource db {
  provider = "postgresql"
}
